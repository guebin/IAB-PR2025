2. if-else

# 예제1 – if문은 if (조건문) 표현식1 else 표현식2의 형태로 실행한다.

x=2
if (x>0) log(x) else print('x는양수이어야함')
[1] 0.6931472
x=0
if (x>0) log(x) else print('x는양수이어야함')
[1] "x는양수이어야함"
x= -1
if (x>0) log(x) else print('x는양수이어야함')
[1] "x는양수이어야함"

- 아래는 잘 실행된다.

if (x>0)    log(x)     else           print('x는양수이어야함')

    print('x는양수이어야함')

2. 함수 check가 올바르게 동작하도록 코드를 수정하라.

check<- function(x){
    if(x%%2==0){
        print("x는 2의 배수")
    }else if(x%%3==0){
        print("x는 3의 배수")
    }else if(x%%6==0){
        print("x는 2의 배수")
        print("x는 3의 배수")
    }else{
        print("x는 3의 배수도 아니고, 2의 배수도 아니다")
    }
}
check(3)
[1] "x는 3의 배수"
check(4)
[1] "x는 2의 배수"
check(5)
[1] "x는 3의 배수도 아니고, 2의 배수도 아니다"
check(6) # 여기가 이상하게 나옴..
[1] "x는 2의 배수"
check(7)
[1] "x는 3의 배수도 아니고, 2의 배수도 아니다"
